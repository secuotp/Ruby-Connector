def getValue(key)
  p = 0
  while p < @keylist.count do
    if @keylist[p] == key
      puts @valuelist[p];
    else
      p += 1
    end
  end
  return nil
end

def peek
  begin
    valueString = Array.new
    valueString[0] = @keylist[@pointer]
    valueString[1] = @valuelist[@pointer]
    return valueString
  rescue

end
